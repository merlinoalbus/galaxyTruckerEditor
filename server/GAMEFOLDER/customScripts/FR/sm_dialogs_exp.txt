SCRIPT ship_random_Dlg

  ShowDlgScene
  ChangeChar tutor campaign/tutor.png
  ShowChar tutor left
  Delay 400
  Say "Désolé. Je ne m'occupe pas des tutoriaux pour l'instant."
  Delay 500
  ChangeChar tutor campaign/tutor-pizza.png
  Delay 500
  Ask "C'est ma pause repas"
  MENU
    OPT_IF tutor_somemissions "Vous ne me reconnaissez pas ?"
      ChangeChar tutor campaign/tutor-smile.png
      Ask "Hé ! Mais c'est [g(mon petit|ma petite)] novice ! Que faites-vous là ?"
      MENU
        OPT "Je regarde juste le contenu de cette nouvelle extension."
          GO show_expansion_stuff
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
    OPT "Je ne m'intéresse pas au tutoriel. Je veux savoir ce qu'il y a de nouveau dans l'extension."
      GO show_expansion_stuff
    END_OF_OPT
  END_OF_MENU

  LABEL show_expansion_stuff
  Delay 300
  ChangeChar tutor campaign/tutor-smile.png
  Delay 200
  Say "Ah, vous allez l'adorer !"
  Say "Les ingénieurs de Corp Inc. ont développé un composant révolutionnaire."
  Ask "Et ils veulent vous faire tester un design de vaisseau non-conventionnel."

  LABEL ship_random_ask
  MENU
    OPT_IFNOT askedHowUnconventional "Euh... à quel point ce design est non-conventionnel ?"
      ChangeChar tutor campaign/tutor-smile.png
      ChangeChar tutor campaign/tutor.png
      Ask "Hé bien, c'est tellement non-conventionnel que vous ne savez pas quelle forme prendra votre vaisseau avant de commencer sa construction."
      SET askedHowUnconventional
      SET showOptCommentOnRandomDesign
      GO ship_random_ask
    END_OF_OPT
    OPT_IF showOptCommentOnRandomDesign "Vous êtes en train de me dire que ce nouveau design est complètement aléatoire ?"
      ChangeChar tutor campaign/tutor.png
      Say "Bien sûr que non !"
      Say "Tout est basé sur un algorithme évolutionnaire sophistiqué qui prend en compte la pression gravitationnelle des 18 pulsars les plus proches, ainsi que les fluctuations quantiques du marché boursier."
      Ask "Les plus brillants esprits de l'ingénierie sanitaire ont bossé là-dessus. Je crois que ça leur a pris toute la nuit pour boucler ça !"
      RESET showOptCommentOnRandomDesign
      GO ship_random_ask
    END_OF_OPT
    OPT_IFNOT askedTellNewComponent "Pouvez-vous m'en dire plus sur ce composant \"révolutionnaire\" ?"
      ChangeChar tutor campaign/tutor.png
      Say "C'est un cargo-container qui peut contenir des cellules d'énergie."
      ChangeChar tutor campaign/tutor.png
      Ask "... ou peut-être un accumulateur d'énergie qui peut contenir des marchandises."
      SET askedTellNewComponent
      SET showOptCommentOnCargoBattery
      GO ship_random_ask
    END_OF_OPT
    OPT_IF showOptCommentOnCargoBattery "Alors le nouveau composant révolutionnaire, c'est juste un container ?"
      ChangeChar tutor campaign/tutor.png
      Say "C'est un container ET un accumulateur d'énergie. Vous n'avez pas entendu ?"
      ChangeChar tutor campaign/tutor.png
      Ask "Peut-être que ça ne semble pas transcendant... mais tout ce qui permet de transporter davantage de matériel dans un vaisseau, c'est forcément bien."
      RESET showOptCommentOnCargoBattery
      GO ship_random_ask
    END_OF_OPT
    OPT "Ca semble bien. C'est parti !"
      ChangeChar tutor campaign/tutor-pizza.png
      Say "Bonne chance, novice."
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU
  HideDlgScene
  RETURN
  RETURN
END_OF_SCRIPT

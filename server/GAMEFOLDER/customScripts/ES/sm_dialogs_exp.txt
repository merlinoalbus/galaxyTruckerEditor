SCRIPT ship_random_Dlg

  ShowDlgScene
  ChangeChar tutor campaign/tutor.png
  ShowChar tutor left
  Delay 400
  Say "Lo siento, no hay tutoriales en activo."
  Delay 500
  ChangeChar tutor campaign/tutor-pizza.png
  Delay 500
  Ask "Es mi pausa para la comida."
  MENU
    OPT_IF tutor_somemissions "¿No me reconoces?"
      ChangeChar tutor campaign/tutor-smile.png
      Ask "¡Vaya, si es mi novat[g(o favorito|a favorita)]! ¿Qué haces tú por aquí?"
      MENU
        OPT "Estoy echando un vistazo a la nueva expansión."
          GO show_expansion_stuff
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
    OPT "No quiero un tutorial. He venido a ver las novedades de la expansión."
      GO show_expansion_stuff
    END_OF_OPT
  END_OF_MENU

  LABEL show_expansion_stuff
  Delay 300
  ChangeChar tutor campaign/tutor-smile.png
  Delay 200
  Say "¡Oh, te va a encantar!"
  Say "Los ingenieros de CoMoSEA han desarrollado un componente revolucionario."
  Ask "Y están deseando que pruebes un diseño de nave muy especial."

  LABEL ship_random_ask
  MENU
    OPT_IFNOT askedHowUnconventional "Ehm... ¿Cómo de “especial”?"
      ChangeChar tutor campaign/tutor-smile.png
      ChangeChar tutor campaign/tutor.png
      Ask "Tan especial que nadie sabe cómo será hasta que empieces a construir la nave."
      SET askedHowUnconventional
      SET showOptCommentOnRandomDesign
      GO ship_random_ask
    END_OF_OPT
    OPT_IF showOptCommentOnRandomDesign "¿Me estás ofreciendo un diseño completamente aleatorio?"
      ChangeChar tutor campaign/tutor.png
      Say "¡Claro que no!"
      Say "Está basado en un sofisticado algoritmo evolutivo que tiene en cuenta la presión gravitónica de los 18 púlsares más cercanos y las fluctuaciones cuánticas del mercado de valores."
      Ask "Algunas de las mentes más brillantes en el ámbito de la ingeniería sanitaria han trabajado en este diseño. ¡Tardaron toda una noche en tenerlo a punto!"
      RESET showOptCommentOnRandomDesign
      GO ship_random_ask
    END_OF_OPT
    OPT_IFNOT askedTellNewComponent "Cuéntame algo más sobre ese “componente revolucionario”."
      ChangeChar tutor campaign/tutor.png
      Say "Es una bodega de carga que puede contener e-pilas..."
      ChangeChar tutor campaign/tutor.png
      Ask "...O una batería que puede contener bodegas de carga. Como más te guste."
      SET askedTellNewComponent
      SET showOptCommentOnCargoBattery
      GO ship_random_ask
    END_OF_OPT
    OPT_IF showOptCommentOnCargoBattery "Vamos, que vuestra gran revolución es una simple bodega de carga."
      ChangeChar tutor campaign/tutor.png
      Say "Una bodega de carga CON e-pilas. ¿Es que no me escuchas?"
      ChangeChar tutor campaign/tutor.png
      Ask "Puede que no sea precisamente espectacular, pero te permite almacenar más trastos en la nave y eso siempre es bueno."
      RESET showOptCommentOnCargoBattery
      GO ship_random_ask
    END_OF_OPT
    OPT "Me gusta la idea. Vamos a probar."
      ChangeChar tutor campaign/tutor-pizza.png
      Say "Buena suerte, novat[g(o|a)]."
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU
  HideDlgScene
  RETURN
  RETURN
END_OF_SCRIPT

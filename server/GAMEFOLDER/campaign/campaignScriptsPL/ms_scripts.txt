SCRIPT ms_rough_mission

  SET firstMissionTaken
  ShowDlgScene
  ShowChar roughtrucker right
  IF wasChanceForBet
    Say "Siema [g(stary|dziewczyno)], jak leci?"
    IF wasBet
      Ask "Jeszcze jeden zakład? Trzy kosmiczne kredyty dla pierwszego pilota na mecie?"
      GO bet_question
    ELSE
      Ask "Zmienił[g(eś|aś)] zdanie? Trzy kosmiczne kredyty dla pierwszego pilota na mecie?"
      GO bet_question
    END_OF_IF
  END_OF_IF
  Ask "Hej [g(stary|dziewczyno)]. Planujesz dołączyć do mnie na tej nieszczęsnej trasie?"
  MENU
    OPT_IFNOT roughintroduced "Cześć. Co tu robisz?"
      SET roughintroduced
      Say "To moja trasa. Kiedy zaczynałem instruktor powiedział mi, że nie jestem dość dobry żeby nią latać."
      Say "I wiesz co? Minęło dziesięć lat, a ja wciąż tu jestem, raz po raz rozwalany przez te przeklęte meteory..."
      Ask "No i kto miał rację, co?"
    END_OF_OPT
    OPT "Brzmi w porządku. Lećmy."
      GO r2a_mission
    END_OF_OPT
    OPT "Chyba poszukam łatwiejszej misji."
      HideChar roughtrucker
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU

  LABEL r2a_mission
  Say "Świetnie, [g(stary|dziewczyno)]. Nawet meteory mogą się znudzić po kilku latach..."
  Ask "A skoro już mowa o nudzie, co powiesz na mały zakładzik? Trzy kosmiczne kredyty dla pierwszego pilota na mecie."
  SET wasChanceForBet

  LABEL bet_question
  RESET triedRaise
  MENU
    OPT "Dobra, czemu nie."
      Say "Jasne! Lećmy!"
      SET wasBet
      GO after_bet
    END_OF_OPT
    OPT_IFNOT triedRaise "Czemu tak mało? Załóżmy się o 30 kredytów!"
      SET triedRaise
      Say "Hej, zdajesz sobie sprawę jak ciężko jest zarobić 30 kredytów?"
      Ask "[g(Stary|Dziewczyno)], zostańmy przy trzech kredytach!"
    END_OF_OPT
    OPT_IFNOT triedRaise "Nigdy się nie zakładam."
      RESET wasBet
      Say "A to szkoda. No nic. Lećmy."
      GO after_bet
    END_OF_OPT
    OPT_IF triedRaise "Pfff! Zapomnij."
      RESET wasBet
      RESET triedRaise
      Say "Dobra, mniejsza o zakład, lećmy."
      Say "(Uff... [g(Ten|Ta)] to ma tylko pieniądze w głowie...)"
      GO after_bet
    END_OF_OPT
  END_OF_MENU

  LABEL after_bet
  HideChar roughtrucker
  HideDlgScene
  ACT_MISSION roughMission
END_OF_SCRIPT

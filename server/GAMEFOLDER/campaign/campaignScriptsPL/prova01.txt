MISSION pippoMission
  AddOpponent roughTrucker
  SetDeckPreparationScript meteor1Deck
  SetSpecCondition bet
  INIT_BUILD
    BuildingHelpScript 500 tutorialAdditionalConcepts
    RESET shouldShowFlightTutorial
    ModifyOpponentsBuildSpeed 75
    SetShipType STI
    SUB_SCRIPT stdParts
  START_BUILDING
  END_BUILDING
  INIT_FLIGHT
  START_FLIGHT
    FlightHelpScript tutorialAdditionalFlightConcepts
  EVALUATE_FLIGHT
    ShowDlgScene
    ShowChar roughtrucker left
    Delay 500
    IF wasBet
      IFMISSIONRESULTIS 1
        Say "O rany."
        Say "Hazard mnie zrujnuje."
        AddMissionCredits 3
        AddOpponentsCredits 1 -3
        Announce "Pilot, z którym się założył[g(eś|aś)], dał ci 3 kredyty."
        UnlockAchievement camp_bet
      ELSE
        IFMISSIONRESULTIS 0
          Say "Wygrałem. Czas wyskakiwać z kasy, [g(koleś|panienko)]."
          AddMissionCredits -3
          AddOpponentsCredits 1 3
          Announce "Musisz zapłacić pilotowi 3 kredyty."
        ELSE
          Say "I to by było na tyle, jeśli idzie o wyścig. [g(Obaj|Oboje)] odpadliśmy."
          Say "To chyba oznacza, że żad[g(en|ne) z nas nie wygrał[g(|o)]."
          Say "Może następnym razem."
        END_OF_IF
      END_OF_IF
    ELSE
      IFMISSIONRESULTIS 1
        Say "Uff, cieszę się, że nie chciał[g(eś|aś)] się zakładać."
      ELSE
        IFMISSIONRESULTIS 0
          Say "Ha, wygrałem! Szkoda, że się nie założyliśmy..."
        ELSE
          Say "I to by było na tyle, jeśli idzie o wyścig. [g(Obaj|Oboje)] odpadliśmy."
        END_OF_IF
      END_OF_IF
    END_OF_IF
    Delay 300
    HideChar roughtrucker
    Delay 200
    HideDlgScene
  END_FLIGHT
FINISH_MISSION
END_OF_MISSION

END_OF_SCRIPTS

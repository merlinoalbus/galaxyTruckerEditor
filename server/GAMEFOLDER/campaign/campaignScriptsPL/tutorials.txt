SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "Cześć, żółtodziobie. Będę twoim instruktorem lotniczym. (stuknij gdziekolwiek, aby kontynuować)"
    Ask "Więc… jak dużo wiesz o pilotowaniu kosmicznej ciężarówki? (wybierz odpowiedź)"
  ELSE
    Say "Cześć, żółtodziobie. Będę twoim instruktorem lotniczym."
    Ask "Więc… jak dużo wiesz o pilotowaniu kosmicznej ciężarówki?"
  END_OF_IF
  MENU
    OPT "Zupełnie nic."
      Say "Nie bój się."
      Say "Wyzysk i Spółka wynajął mnie z myślą o jednym: zrobieniu z ciebie galaktycznego pilota ciężarówki pierwszej klasy."
      Ask "Przygotowaliśmy dla ciebie fajny samouczek."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "Znam wersję planszową, ale wersję cyfrową widzę po raz pierwszy."
      Say "Ach, w takim razie to będzie bułka z masłem. Zasady są praktycznie identyczne."
      Ask "A my przygotowaliśmy dla ciebie samouczek."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "Wiem już wszystko. Po prostu pozwól mi grać!"
      Say "Aha… Jesteś [g(jednym|jedną)] z tych…"
      IF_TUTORIAL_SEEN
        Say "(Raz widział[g(|a)] samouczek i już myśli, że wszystko wie.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Przeleć się wahadłowcem do stacji kosmicznej Port Żółtodziób. Tam się spotkamy."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Fajnie. Jestem got[g(ów|owa)]."
      Say "Dobrze."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Cóż, widziałem już samouczek, ale drugi raz mi chyba nie zaszkodzi."
      Say "Dobry wybór."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Dzięki, ale już widział[g(em|am)] samouczek."
      Say "Aha, w porządku."
      Say "(Wiedziałem, że skądś cię znam.)"
      Say "W takim razie polecimy od razu na stację kosmiczną Port Żółtodziób."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "Nie cierpię samouczków. Po prostu pozwól mi grać."
      Say "Jesteś pew[g(ien|na)]?"
      Say "To najlepszy sposób na poznanie podstaw rozgrywki."
      Say "Będziemy mieć okazję polecieć razem do stacji kosmicznej Port Żółtodziób."
      Ask "Będzie fajnie."
      MENU
        OPT "OK, zróbmy to."
          Say "Dobry wybór."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "Nie, dzięki."
          SET tutorialRefused
          Say "Twoja strata."
          Say "Przeleć się wahadłowcem do stacji kosmicznej Port Żółtodziób. Tam się spotkamy."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT

SCRIPT tutorialIntro

  ShowDlgScene
  Delay 500
  ShowChar tutor right
  Delay 500
  IF_FROM_CAMPAIGN
    Say "Witaj w magazynie Wyzysku i Spółki."
    Say "Tutaj będziesz budować swój statek."
  ELSE
    Say "Witaj. Będę twoim instruktorem lotniczym. (stuknij gdziekolwiek, aby kontynuować)"
    Ask "Czy jesteś got[g(ów|owa)] poznać podstawy pilotowania kosmicznej ciężarówki? (wybierz odpowiedź)"
    MENU
      OPT "Tak jest, proszę pana!"
        Say "Podoba mi się twój zapał!"
        EXIT_MENU
      END_OF_OPT
      OPT "Niby tak…"
        Say "Trochę więcej entuzjazmu! Będziesz budować prawdziwy statek kosmiczny."
        EXIT_MENU
      END_OF_OPT
      OPT "Yyy… nie chciałem tutaj trafić. Mogę stąd wyjść?"
        Say "Ech…"
        Delay 300
        Delay 300
        Say "Hej, chłopaki, trafił[g(|a)] nam się kolejn[g(y|a)] przybłęda. Wyrzućcie [g(go|ją)]."
        HideChar tutor
        HideDlgScene
        QuitCampaign
        RETURN
      END_OF_OPT
    END_OF_MENU
  END_OF_IF
  Delay 300
  ChangeChar tutor campaign/tutor-out.png
  Delay 400
  Say "Hej, staruszku! Pokaż nam infografikę!"
  Delay 300
  ChangeChar tutor campaign/tutor.png
  Delay 300
  HideChar tutor
  HideDlgScene
  AddInfoWindow "tutorial/infoGameGoal.png"
  AddInfoWindow "tutorial/infoGameGoal2.png"
  AddInfoWindow "tutorial/infoGameGoal3.png"
  AddInfoWindow "tutorial/infoGameGoal4.png"
  ShowInfoWindow "tutorial/infoGameGoal.png"
  ShowDlgScene
  ChangeChar tutor campaign/tutor.png
  ShowChar tutor right
  Say "Proste, nie? Zaczynamy."
  Say "Będę miał na ciebie oko, w razie gdybyś potrzebował[g(|a)] pomocy."
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
END_OF_SCRIPT

SCRIPT tutorialMeteorFinal

  ShowDlgScene
  Delay 200
  ShowChar tutor left
  Delay 200
  Say "Czy rozumiesz już jak działają meteory?"
  RESET playerPaysAttention
  MENU
    OPT "Tak, jak najbardziej."
      Say "Świetnie."
      EXIT_MENU
    END_OF_OPT
    OPT "Nie bardzo, ale zawsze mogę jeszcze raz przeczytać treść karty."
      Say "Świetnie."
      SET playerPaysAttention
      EXIT_MENU
    END_OF_OPT
    OPT "Czy mógłbyś to jakoś podsumować?"
      Say "Jasne. MAŁE meteory obijają się od twojego statku, możesz też zatrzymać je osłoną. DUŻE meteory może zatrzymać tylko ostrzał z działa znajdującego się w kolumnie/rzędzie, z której nadlatują."
      EXIT_MENU
    END_OF_OPT
  END_OF_MENU
  Delay 500
  Say "OK. Myślę, że poradzisz sobie z resztą przygód bez mojej pomocy."
  IFNOT playerPaysAttention
    Say "Nie zapomnij przeczytać treści kart, których nie znasz."
  END_OF_IF
  Say "Do zobaczenia na mecie!"
  HideChar tutor
  HideDlgScene
END_OF_SCRIPT

SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "Hola, novat[g(o|a)]. A partir de ahora seré tu instructor de vuelo.                                                                          (Pulsa en cualquier lugar para continuar.)"
    Ask "Vamos a ver... ¿Qué sabes de este juego?                                            (Elige una respuesta.)"
  ELSE
    Say "Hola, novat[g(o|a)]. A partir de ahora seré tu instructor de vuelo."
    Ask "Vamos a ver... ¿Qué sabes de este juego?"
  END_OF_IF
  MENU
    OPT "No tengo ni idea."
      Say "No te preocupes."
      Say "CoMoSEA me ha contratado para que haga de ti [g(un camionero|una camionera)] profesional."
      Ask "He preparado un tutorial estupendo para ti."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "Sé cómo funciona el juego de mesa, pero es la primera vez que pruebo la versión digital."
      Say "Oh, entonces no habrá problema. Las reglas son prácticamente las mismas."
      Ask "Además, tengo preparado un tutorial estupendo para ti."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "¿Empezamos de una vez? Ya sé jugar."
      Say "Qué bien, otr[g(o|a)] listill[g(o|a)]."
      IF_TUTORIAL_SEEN
        Say "(Ha terminado el tutorial y se cree que lo sabe todo.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Sube a la lanzadera para ir a Puerto Novato. Nos veremos allí."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Estoy list[g(o|a)] para el tutorial."
      Say "Bien."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Ya he visto el tutorial, pero puedo repetirlo si quieres."
      Say "Buena elección."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Gracias, pero ya he visto el tutorial."
      Say "Ah, muy bien."
      Say "(Por eso me resultabas tan familiar...)"
      Say "Entonces podemos ir directamente a Puerto Novato."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "No me gustan los tutoriales. Deja que me lo salte."
      Say "¿Estás segur[g(o|a)]?"
      Say "Es la mejor manera de aprender a jugar."
      Say "Además, podremos volar juntos hasta Puerto Novato."
      Ask "Será divertido."
      MENU
        OPT "Vale, vamos a hacer el tutorial..."
          Say "Buena elección."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "No, gracias."
          SET tutorialRefused
          Say "Tú te lo pierdes."
          Say "Sube a la lanzadera para ir a Puerto Novato. Nos veremos allí."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT
  SCRIPT tutorialIntro
    ShowDlgScene
    Delay 500
    ShowChar tutor right

      Say "Bienvenido al almacén de CoMoSEA."

      Ask "¿Estás list[g(o|a)] para aprender las bases del juego?                                            (Elige una respuesta.)"
      MENU
        OPT "¡Señor, sí, señor!"
          Say "¡Así me gusta!"
          EXIT_MENU

          Delay 300
          Delay 300
          Say "¡Chicos! ¡Tenemos a otr[g(o|a)] de es[g(os|as)]! ¡Sacadl[g(o|a)] de aquí!"
          HideChar tutor
          HideDlgScene
          QuitCampaign
          RETURN
        END_OF_OPT
      END_OF_MENU
    END_OF_IF
    Delay 300
    ChangeChar tutor campaign/tutor-out.png
    Delay 400
    Say "Oye, abuelo. Saca las infografías."
    Delay 300
    ChangeChar tutor campaign/tutor.png

SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "A hele, další [g(začátečník|začátečnice)]. Vítám tě, budu tvůj letový instruktor.                                 ([v(ťap|klik)]ni kamkoliv)"
    Ask "Takže... Kolik toho víš o truckerech a o létání skrz galaxii?                                         (vyber odpověď)"
  ELSE
    Say "A hele, další [g(začátečník|začátečnice)]. Vítám tě, budu tvůj letový instruktor."
    Ask "Takže... Kolik toho víš o truckerech a o létání skrz galaxii?"
  END_OF_IF
  MENU
    OPT "Vlastně vůbec nic"
      Say "Neboj."
      Say "Společnost s.r.o. mě najala, abych tě vyškolil na prvotřídního truckera."
      Ask "Máme také pěknou výukovou misi."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "Znám to z deskové verze, ale tuhle digitální zatím neznám."
      Say "Ach, pak to bude jednoduché. Pravidla jsou v podstatě stejná."
      Ask "A také máme pěknou výukovou misi."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "Prostě mě nechej hrát, všechno vím."
      Say "Aha... Další takov[g(ej|á)]."
      IF_TUTORIAL_SEEN
        Say "(Jen proto, že už viděl[g(|a)] výukovou misi, si myslí, že všechno zná.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Tak zatím. Uvidíme se na stanici Lama."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Bezva. Těším se na výukovou misi."
      Say "Prima."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Už jsem výukovou misi zkoušel[g(|a)]. Ale co, kouknu na ni znovu."
      Say "Dobrá volba."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Díky, ale už jsem výukovou misi proš[g(el|la)]."
      Say "Aha, dobrá."
      Say "(Jsem si říkal, že tě odněkud znám.)"
      Say "V tom případě pojďme rovnou na stanici Lama."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "Nesnáším výukové mise. Prostě mě nech hrát."
      Say "A jsi si tím jist[g(ý|á)]?"
      Say "Je to nejlepší cesta, jak se naučit hru."
      Say "A navíc bychom se spolu proletěli na stanici Lama."
      Ask "Byla by to sranda."
      MENU
        OPT "No tak dobře, pojďme na to."
          Say "Dobrá volba."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "Fakt ne, díky."
          SET tutorialRefused
          Say "No jak myslíš."
          Say "Uvidíme se na stanici Lama."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT
  SCRIPT tutorialIntro
    ShowDlgScene
    Delay 500
    ShowChar tutor right
    Delay 500
    IF_FROM_CAMPAIGN
      Say "Vítej ve skladišti Společnosti s.r.o."
      Say "Tady se staví lodě."
    ELSE
      Say "Vítám tě. Budu tvůj letový instruktor.              ([v(ťap|klik)]ni kamkoliv)"
      Ask "Jsi připraven[g(|a)] naučit se vše, co má správný trucker znát?"
      MENU
        OPT "Rozkaz, pane!"
          Say "Tomu říkám nadšení!"
          EXIT_MENU
        END_OF_OPT
        OPT "Pro mě za mě..."
          Say "No tak, projev trochu nadšení. Budeš stavět skutečnou kosmickou loď!"
          EXIT_MENU

          Say "Ach jo."
          Delay 300
          Delay 300
          Say "Hej, chlapi, máme tady [g(dalšího|další takovou)]. Vyhoďte [g(ho|ji)]."
          HideChar tutor
          HideDlgScene
          QuitCampaign
          RETURN
        END_OF_OPT
      END_OF_MENU
    END_OF_IF
    Delay 300
    ChangeChar tutor campaign/tutor-out.png
    Delay 400
    Say "Hej, staříku! Přihrej nám sem nějakou infografiku."
    Delay 300
    ChangeChar tutor campaign/tutor.png
    Delay 300
    HideChar tutor
    HideDlgScene
    AddInfoWindow tutorial/infoGameGoal.png
    AddInfoWindow tutorial/infoGameGoal2.png
    AddInfoWindow tutorial/infoGameGoal3.png
    AddInfoWindow tutorial/infoGameGoal4.png
    ShowInfoWindow "tutorial/infoGameGoal.png"
    ShowDlgScene
    ChangeChar tutor campaign/tutor.png
    ShowChar tutor right
    Say "Jednoduché, ne?"
    Say "Budu na tebe dávat pozor, kdybys potřeboval[g(|a)] pomoc."
    Delay 300
    HideChar tutor
    Delay 300
    HideDlgScene
  END_OF_SCRIPT

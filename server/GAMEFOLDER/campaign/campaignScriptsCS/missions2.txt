SCRIPT pirateEscape

  IF pirateEscaped
    RETURN
  END_OF_IF
  SET pirateEscapedAtLeastOnce
  SET pirateEscaped
  ShowDlgScene
  Announce "Než budete pokračovat, zkontrolovali jste pirátovu celu... Je prázdná!"
  IfMissionResultIs 1
    Announce "Asi se mu podařilo vyklouznout z lodi, zatímco jste nakládali zboží."
  END_OF_IF
  IfMissionResultIs 2
    Announce "Asi se mu podařilo vmísit se mezi posádku, co opouštěla tvou loď."
  END_OF_IF

    Announce "Asi se mu podařilo vmísit se mezi posádku, co šla vyrabovat opuštěnou stanici – a nevrátil se zpět."
  END_OF_IF

  END_OF_IF
  IfMissionResultIs 5
    Announce "Asi se mu podařilo nějak se vmísit mezi členy posádky, co nepřátelé unesli."
  END_OF_IF
  IfMissionResultIs 6
    Announce "Asi se schoval do jedné z beden, které nepřátelé odnesli."
  END_OF_IF
  IfMissionResultIs 8
    Announce "Asi se nějak schoval mezi baterie, které nepřátelé odnesli."
  END_OF_IF
  IfMissionResultIs 7
    Announce "Asi se vmísil mezi posádku, co čistila špinavé konektory... A pak... Jak se mu sakra podařilo utéct?"
    Announce "(Příště by to chtělo nemít žádné otevřené přípojky.)"
  END_OF_IF
  HidePath U2-17x
  IFNOT ropeCut
    HideButton bpirate3
  END_OF_IF

    OPT "Sakryš, vrátíme se do pirátské skrýše."

    END_OF_OPT

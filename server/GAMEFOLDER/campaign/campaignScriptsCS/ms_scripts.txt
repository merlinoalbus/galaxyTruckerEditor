SCRIPT ms_rough_mission

  SET firstMissionTaken
  ShowDlgScene
  ShowChar roughtrucker right
  IF wasChanceForBet
    Say "Vítej zpět, [g(kámo|holka)]."
    IF wasBet
      Ask "Chceš se zase vsadit? Tři vesmírný babky pro toho, kdo doletí první?"
      GO bet_question
    ELSE
      Ask "Tak co, nezměnil[g(|a)] jsi názor? Tři vesmírný babky pro toho, kdo doletí první?"
      GO bet_question
    END_OF_IF
  END_OF_IF
  Ask "Nazdar, [g(kámo|ženská)]. Chceš se ke mě připojit na týhle bohama zatracený štrece?"
  MENU
    OPT_IFNOT roughintroduced "Nazdar. Co tu děláš?"
      SET roughintroduced
      Say "Jezdím tuhle štreku. Když jsem začínal, můj instruktor mi řekl, že na to nemám."
      Say "A vidíš? Jsem tady a kličkuju mezi těma pitomejma meteorama znovu a znovu, už 10 let."
      Ask "Tak kdo měl pravdu, co?"
    END_OF_OPT
    OPT "Dobrá, letíme."
      GO r2a_mission
    END_OF_OPT
    OPT "Asi se radši podívám po jednodušší misi."
      HideChar roughtrucker
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU

  LABEL r2a_mission
  Say "Výborně, [g(kámo|holka)]. Víš, tyhle meteory začnou jednoho nudit, po pár letech."
  Ask "Když už mluvíme o nudě, co takhle sázka? Kdo doletí první, dostane od toho druhýho 3 kosmický babky."
  SET wasChanceForBet

  LABEL bet_question
  RESET triedRaise
  MENU
    OPT "Dobrá, proč ne."
      Say "Bezva. Tak jdem na to."
      SET wasBet
      GO after_bet
    END_OF_OPT
    OPT_IFNOT triedRaise "Proč tak málo? Zvedněme to na 30!"
      SET triedRaise
      Say "Hele, tušíš ty vůbec, jak dlouho trvá vydělat 30 babek?"
      Ask "Zůstaňme na třech, [g(kámo|holka)]."
    END_OF_OPT
    OPT_IFNOT triedRaise "Nikdy se nesázím."
      RESET wasBet
      Say "Tak to je škoda, no. Neva. Jdeme na to."
      GO after_bet
    END_OF_OPT
    OPT_IF triedRaise "Pche, tak v tom případě na to zapomeň."
      RESET wasBet
      RESET triedRaise
      Say "Dobrá, v tom případě teda bez sázky."
      Say "(Uff... [g(Ten|Ta)] má asi peněz, že neví co s nima...)"
      GO after_bet
    END_OF_OPT
  END_OF_MENU

  LABEL after_bet
  HideChar roughtrucker
  HideDlgScene
  ACT_MISSION roughMission
END_OF_SCRIPT

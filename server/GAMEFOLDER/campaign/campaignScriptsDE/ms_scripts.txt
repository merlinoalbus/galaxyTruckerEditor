SCRIPT ms_rough_mission

  SET firstMissionTaken
  ShowDlgScene
  ShowChar roughtrucker right
  IF wasChanceForBet
    Say "He, [g(Kumpel|Kollegin)], wie läuft’s denn so?"
    IF wasBet
      Ask "Willst du noch einmal wetten? 3 kosmische Credits für den Trucker, der es zuerst schafft."
      GO bet_question
    ELSE
      Ask "Hast du deine Meinung geändert? 3 kosmische Credits für den Trucker, der es zuerst schafft."
      GO bet_question
    END_OF_IF
  END_OF_IF
  Ask "He, [g(Kumpel|Kollegin)], machst du bei dieser üblen Strecke mit?"
  MENU
    OPT_IFNOT roughintroduced "Hallo. Was machst du denn hier?"
      SET roughintroduced
      Say "Ich trucke auf dieser Strecke. Als ich angefangen habe, hat mir mein Ausbilder gesagt, ich wäre nicht gut genug dafür."
      Say "Und siehe da: Jetzt bin ich hier und werde seit 10 Jahren immer wieder von diesen Meteoriten zerlegt."
      Ask "Wer hatte also recht, hä?"
    END_OF_OPT
    OPT "Klingt nach Spaß. Ich fliege mit."
      GO r2a_mission
    END_OF_OPT
    OPT "Ich denke, ich suche mir lieber eine einfachere Mission."
      HideChar roughtrucker
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU

  LABEL r2a_mission
  Say "Super, [g(Kumpel|Kollegin)]! Diese Meteoriten werden auf Dauer auch langweilig."
  Ask "Wo ich gerade von Langeweile spreche: wie wäre es mit einer Wette? Wer zuerst ankommt, kriegt 3 kosmische Credits vom anderen."
  SET wasChanceForBet

  LABEL bet_question
  RESET triedRaise
  MENU
    OPT "Na gut, warum nicht?"
      Say "Recht so! Los geht’s!"
      SET wasBet
      GO after_bet
    END_OF_OPT
    OPT_IFNOT triedRaise "Warum so wenig? Erhöhen wir auf 30 Credits."
      SET triedRaise
      Say "Hast du ’ne Ahnung, wie schwer es ist, 30 Kröten zu verdienen?"
      Ask "Lass uns bei 3 bleiben, [g(Kumpel|Kollegin)]."
    END_OF_OPT
    OPT_IFNOT triedRaise "Ich mache keine Glücksspiele."
      RESET wasBet
      Say "Schade. Na gut. Legen wir los!"
      GO after_bet
    END_OF_OPT
    OPT_IF triedRaise "Pah! Dann vergiss es."
      RESET wasBet
      RESET triedRaise
      Say "Na gut, lass uns ohne Wette fliegen."
      Say "(Puh! Geldgeil und bekloppt.)"
      GO after_bet
    END_OF_OPT
  END_OF_MENU

  LABEL after_bet
  HideChar roughtrucker
  HideDlgScene
  ACT_MISSION roughMission
END_OF_SCRIPT

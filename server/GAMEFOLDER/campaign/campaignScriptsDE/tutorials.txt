SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "Hallo [g(Anfänger|Anfängerin)]. Ich bin dein Flugausbilder. (Zum Fortfahren irgendwo [v(tipp|klick)]en.)"
    Ask "Also ... was weißt du denn bereits über das Galaxy Trucking? (Wähle eine Antwort.)"
  ELSE
    Say "Hallo [g(Anfänger|Anfängerin)]. Ich bin dein Flugausbilder."
    Ask "Also ... was weißt du denn bereits über das Galaxy Trucking?"
  END_OF_IF
  MENU
    OPT "Überhaupt nichts."
      Say "Keine Sorge."
      Say "Die Corporation mbH hat mich gebeten, aus dir einen erstklassigen Galaxy Trucker zu machen."
      Ask "Wir haben ein hübsches Tutorial vorbereitet."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "Die Brettspielversion kenne ich, aber dieser Digitalkram ist Neuland für mich."
      Say "Och, dann wird das ein Kinderspiel. Die Spielregeln sind fast gleich."
      Ask "Und wir haben ein Tutorial für dich."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "Lass mich einfach loslegen. Ich weiß alles!"
      Say "Oh ... schon wieder so [g(einer|eine)]."
      IF_TUTORIAL_SEEN
        Say "([g(Er|Sie)] hat das Tutorial gesehen und denkt jetzt, dass [g(er|sie)] alles weiß.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Nimm die Fähre zur Neulingshafen-Raumstation. Dort treffen wir uns wieder."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Cool. Ich bin bereit für das Tutorial."
      Say "Gut."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Also, das Tutorial habe ich zwar schon gesehen. Aber ich mache es ruhig noch einmal."
      Say "Gute Wahl."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Danke, aber ich habe das Tutorial bereits gesehen."
      Say "Oh, okay."
      Say "(Du kamst mir doch gleich so bekannt vor.)"
      Say "Dann lass uns sofort zur Neulingshafen-Raumstation gehen."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "Ich finde Tutorials blöd. Lass mich einfach loslegen."
      Say "Bist du sicher?"
      Say "Das ist nämlich die beste Art, das Spiel zu erlernen."
      Say "Außerdem fliegen wir dann zusammen zur Neulingshafen-Raumstation."
      Ask "Das wird bestimmt ein Sapß."
      MENU
        OPT "Oh, na gut. Dann lass uns loslegen."
          Say "Gute Antwort."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "Nein danke."
          SET tutorialRefused
          Say "Schade drum."
          Say "Nim eine Fähre nach Neulingshafen. Dort treffen wir uns wieder."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT

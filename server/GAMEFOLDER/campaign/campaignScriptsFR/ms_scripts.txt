SCRIPT ms_rough_mission

  SET firstMissionTaken
  ShowDlgScene
  ShowChar roughtrucker right
  IF wasChanceForBet
    Say "Hé [g(l'ami|l'amie)], comment ça va ?"
    IF wasBet
      Ask "Vous voulez refaire un pari ? 3 Crédits au premier pilote sur la ligne d'arrivée, ok ?"
      GO bet_question
    ELSE
      Ask "Alors, vous avez changé d'avis ? 3 Crédits au premier pilote sur la ligne d'arrivée, ok ?"
      GO bet_question
    END_OF_IF
  END_OF_IF
  Ask "Hé [g(l'ami|l'amie)], ça vous dit de m'accompagner pour ce triste vol ?"
  MENU
    OPT_IFNOT roughintroduced "Salut. Que faites-vous là ?"
      SET roughintroduced
      Say "Ça fait un moment que je suis sur cette route. Quand j'ai commencé, mon instructeur m'avait dit que je n'avais pas le niveau."
      Say "Et vous voyez ? Me voilà, martelé encore et encore par ces innombrables météorites depuis 10 longues années."
      Ask "Alors, qui avait raison, hein ?"
    END_OF_OPT
    OPT "Ça a l'air sympa. Je volerai à vos côtés."
      GO r2a_mission
    END_OF_OPT
    OPT "Je pense que je vais chercher une mission plus facile."
      HideChar roughtrucker
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU

  LABEL r2a_mission
  Say "Super [g(l'ami|l'amie)]. Ces météorites ont tendance à être vraiment ennuyeuses avec les années."
  Ask "En parlant d'ennui, que diriez-vous de pimenter les choses ? Le premier à franchir la ligne d'arrivée reçoit 3 Crédits du deuxième."
  SET wasChanceForBet

  LABEL bet_question
  RESET triedRaise
  MENU
    OPT "Ok, pourquoi pas ?"
      Say "Parfait ! Allons-y !"
      SET wasBet
      GO after_bet
    END_OF_OPT
    OPT_IFNOT triedRaise "Vous êtes petit joueur ? Partons plutôt sur 30 Crédits."
      SET triedRaise
      Say "Eh, vous savez combien il est difficile de se faire 30 billets ?"
      Ask "Restons sur 3, [g(l'ami|l'amie)] !"
    END_OF_OPT
    OPT_IFNOT triedRaise "Je ne fais jamais de pari."
      RESET wasBet
      Say "Quel dommage. Mais bon, allons-y."
      GO after_bet
    END_OF_OPT
    OPT_IF triedRaise "Bah ! Laissez tomber."
      RESET wasBet
      RESET triedRaise
      Say "Ok, oublions le pari et allons-y."
      Say "(Ouf ! C'est [g(un vrai profiteur|une vrai profiteuse].)"
      GO after_bet
    END_OF_OPT
  END_OF_MENU

  LABEL after_bet
  HideChar roughtrucker
  HideDlgScene
  ACT_MISSION roughMission
END_OF_SCRIPT

SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "Salut, novice. Je suis ici pour être votre instructeur. 
 (Cliquez n'importe où pour continuer.)"
    Ask "Dites-moi... que connaissez-vous du transport spatial ? 
 (Choisissez une réponse.)"
  ELSE
    Say "Salut, novice. J'suis ici pour être votre instructeur."
    Ask "Dites-moi... que connaissez-vous du transport spatial ?"
  END_OF_IF
  MENU
    OPT "Rien du tout."
      Say "Ne vous inquiétez pas."
      Say "Corporation Incorporated m'a embauché pour faire de vous un pilote de première classe."
      Ask "Nous avons un tutoriel tout indiqué pour vous."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "Je connais le jeu de plateau, mais la version numérique est nouvelle pour moi."
      Say "Ah, je vois, ça va être du gâteau. Les règles sont pratiquement les mêmes."
      Ask "Et nous avons un tutoriel fait pour vous."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "Laissez-moi faire. Je connais déjà tout !"
      Say "Allons bon... encore un."
      IF_TUTORIAL_SEEN
        Say "([g(Il|Elle)] a juste suivi le tutoriel et [il] pense déjà tout savoir.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Prenez la navette pour la station spatiale de Port Novice. Je vous retrouve là-bas."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Super. Je suis prêt pour ce cours."
      Say "Parfait."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Bon, j'ai déjà suivi ce tutoriel. Mais je veux bien le refaire."
      Say "Voilà un choix raisonnable."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Merci, mais j'ai déjà suivi ce cours."
      Say "Ah, très bien."
      Say "(Je me disais bien vous avoir déjà vu quelque part.)"
      Say "Alors, rendez-vous à la station spatiale de Port Novice."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "Je déteste les tutoriels. Laissez-moi jouer."
      Say "Êtes-vous sûr ?"
      Say "C'est pourtant la meilleure façon pour apprendre le jeu."
      Say "Et cela nous permettra de faire le voyage jusqu'à la station spatiale de Port Novice ensemble."
      Ask "Ça sera sympa."
      MENU
        OPT "Je vois, très bien. Allons-y."
          Say "Bonne réponse."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "Non merci."
          SET tutorialRefused
          Say "Tant pis pour vous."
          Say "Prenez une navette pour Port Novice. Je vous retrouve là-bas."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT

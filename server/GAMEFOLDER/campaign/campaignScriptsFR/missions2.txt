SCRIPT pirateEscape

  IF pirateEscaped

  END_OF_IF

  ShowDlgScene

  IfMissionResultIs 1
    Announce "Il a dû profiter de ce que vous chargiez des marchandises pour s'échapper."
  END_OF_IF
  IfMissionResultIs 2
    Announce "Apparemment, il s'est faufilé parmi les membres d'équipage qui ont quitté votre vaisseau."
  END_OF_IF
  IfMissionResultIs 3
    Announce "Apparemment, il est sorti avec l'équipage parti piller la station abandonnée… et il n'est pas revenu !"
  END_OF_IF

    Announce "Il a dû filer en douce alors que vous récupériez des marchandises des contrebandiers."
  END_OF_IF

    Announce "Il devait faire partie des membres d'équipage que les méchants ont emportés."
  END_OF_IF
  IfMissionResultIs 6
    Announce "Il s'est sûrement caché dans l'une des caisses que les méchants vous ont prises."
  END_OF_IF
  IfMissionResultIs 8
    Announce "Il a dû se débrouiller pour se cacher parmi les cellules d'énergie que les méchants vous ont volées."
  END_OF_IF
  IfMissionResultIs 7
    Announce "Il est probablement sorti avec l'équipe de maintenance des connecteurs et… où est-il passé ensuite ?"
    Announce "(La prochaine fois, nous ferions mieux de voler sans aucun connecteur exposé.)"

  HidePath U2-17x
  IFNOT ropeCut
    HideButton bpirate3
  END_OF_IF
  ShowButton bpirate2
  SetFocus bpirate2
  MENU
    OPT "Nous allons lui apprendre à s'évader ainsi ! Cap sur le Repaire des Pirates."
      GiveUpFlight
      EXIT_MENU
    END_OF_OPT
    OPT "Et puis zut. Continuons vers Administra sans lui."

    END_OF_OPT

    END_OF_IF
    IF_IS destination 1
      Say "Robot-minier AX Gamma, terminé."
      HideChar tradeBot1
    END_OF_IF
    IF_IS destination 2
      Say "Robot-minier AX Bêta, terminé."
      HideChar tradeBot2
    END_OF_IF

  ELSE

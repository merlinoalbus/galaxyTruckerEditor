SCRIPT campaignIntro

  ShowDlgScene
  Delay 1000
  ShowChar tutor left
  Delay 1000
  RESET tutorialSeen
  IF_TUTORIAL_SEEN
    SET tutorialSeen
  END_OF_IF
  IFNOT tutorialSeen
    Say "Hi, newbie. I'm here to be your flight instructor.                                                                          ([v(Tap|Click)] anywhere to continue.)"
    Ask "So ... how much do you know about galaxy trucking?                                            (Choose an answer.)"
  ELSE
    Say "Hi, newbie. I'm here to be your flight instructor."
    Ask "So ... how much do you know about galaxy trucking?"
  END_OF_IF
  MENU
    OPT "Nothing at all."
      Say "Don't worry."
      Say "Corporation Incorporated has hired me to turn you into a first-rate galaxy trucker."
      Ask "We have a nice tutorial for you."
      SET playerIsBeginner
      EXIT_MENU
    END_OF_OPT
    OPT "I know all about cardboard, but this digital version is new to me."
      Say "Oh, this will be a piece of cake. The rules are pretty much the same."
      Ask "And we have a tutorial for you."
      SET playerIsAdvanced
      EXIT_MENU
    END_OF_OPT
    OPT "Just let me in. I know everything!"
      Say "Oh ... another one."
      IF_TUTORIAL_SEEN
        Say "([g(He|She|It)] already saw the tutorial and now [g(he|she|it)] thinks [g(he|she|it)] knows everything.)"
      END_OF_IF
      SET playerIsCocky
      SET tutorialRefused
      Say "Take a shuttle to Port Newbie space station. I'll meet you there."
      GO campaignIntroEnd
    END_OF_OPT
  END_OF_MENU
  MENU
    OPT "Cool. I'm ready for the tutorial."
      Say "Good."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "We'll, I've seen the tutorial already. But let's try it again."
      Say "Good choice."
      SET tutorialAsked
      EXIT_MENU
    END_OF_OPT
    OPT_IF tutorialSeen "Thanks, but I've already seen the tutorial."
      Say "Oh, okay."
      Say "(I thought I knew you from somewhere.)"
      Say "Let's just go straight to Port Newbie space station, then."
      SET tutorialSkipped
      GO campaignIntroEnd
    END_OF_OPT
    OPT_IFNOT tutorialSeen "I hate tutorials. Just let me in."
      Say "Are you sure?"
      Say "It's the best way to learn the game."
      Say "And we get to fly together to Port Newbie space station."
      Ask "It'll be fun."
      MENU
        OPT "Oh, all right. Let's do it."
          Say "Good answer."
          SET tutorialAsked
          EXIT_MENU
        END_OF_OPT
        OPT "No thanks."
          SET tutorialRefused
          Say "Your loss."
          Say "Take a shuttle to Port Newbie. I'll meet you there."
          GO campaignIntroEnd
        END_OF_OPT
      END_OF_MENU
    END_OF_OPT
  END_OF_MENU

  LABEL campaignIntroEnd
  Delay 300
  HideChar tutor
  Delay 300
  HideDlgScene
  SUB_SCRIPT newbie_init
  IF tutorialAsked
    ACT_MISSION Tutorial
    SetCredits 0
  ELSE
    CenterMapByNode newbie
  END_OF_IF
END_OF_SCRIPT

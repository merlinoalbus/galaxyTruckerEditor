SCRIPT ms_rough_mission

  SET firstMissionTaken
  ShowDlgScene
  ShowChar roughtrucker right
  IF wasChanceForBet
    Say "Эй[g(, братишка|, сестренка)], как оно?"
    IF wasBet
      Ask "Повторим? 3 монеты тому, кто доберется до финиша первым, а?"
      GO bet_question
    ELSE
      Ask "Что, передумал[g(|a)]? 3 монеты тому, кто доберется до финиша первым, а?"
      GO bet_question
    END_OF_IF
  END_OF_IF
  Ask "Эй[g(, братишка|, сестренка)], составишь мне компанию на этом убогом маршруте?"
  MENU
    OPT_IFNOT roughintroduced "Привет. А что ты тут делаешь?"
      SET roughintroduced
      Say "Вожу грузы по этому маршруту. Когда я начинал, мой инструктор сказал мне, что я с ним не справлюсь."
      Say "И что теперь? Вот он я, мотаюсь между этими метеоритами туда-сюда уже 10 лет кряду."
      Ask "Ну и кто теперь прав, а? То-то."
    END_OF_OPT
    OPT "Звучит неплохо. Полетели."
      GO r2a_mission
    END_OF_OPT
    OPT "Думаю, поищу что-нибудь попроще."
      HideChar roughtrucker
      HideDlgScene
      RETURN
    END_OF_OPT
  END_OF_MENU

  LABEL r2a_mission
  Say "Отлично[g(, братишка|, сестренка)]. Эти метеориты могут здорово наскучить за несколько лет."
  Ask "Кстати, про скуку. Как насчет небольшой ставки? Тот, кто прилетит первым, получит 3 космобакса, идет?"
  SET wasChanceForBet

  LABEL bet_question
  RESET triedRaise
  MENU
    OPT "Давай, почему бы и нет?"
      Say "Правильно! Поехали!"
      SET wasBet
      GO after_bet
    END_OF_OPT
    OPT_IFNOT triedRaise "А чего так вяло? Давай увеличим до 30, зачем мелочиться?"
      SET triedRaise
      Say "Эй, да ты вообще знаешь, как сложно заработать эти 30 монет?"
      Ask "Давай остановимся на 3, а[g(, братишка|, сестренка)]?"
    END_OF_OPT
    OPT_IFNOT triedRaise "Не имею привычки делать ставки."
      RESET wasBe
      Say "Обидно, слушай. Ну и ладно. Покатили."
      GO after_bet
    END_OF_OPT
    OPT_IF triedRaise "Вот еще! Тогда забыли."
      RESET wasBet
      RESET triedRaise
      Say "Хорошо, давай полетим просто так."
      Say "(Похоже, [g(этот|эта)] опять из тех, что не знают цену деньгам.)"
      GO after_bet
    END_OF_OPT
  END_OF_MENU

  LABEL after_bet
  HideChar roughtrucker
  HideDlgScene
  ACT_MISSION roughMission
END_OF_SCRIPT

SCRIPT stdParts

  IF purpleUnlocked
    IF brownUnlocked
      AddShipParts "parts/allParts.yaml"
    ELSE
      AddShipParts "parts/noBrown.yaml"
    END_OF_IF
  ELSE
    AddShipParts "parts/noAliens.yaml"
  END_OF_IF

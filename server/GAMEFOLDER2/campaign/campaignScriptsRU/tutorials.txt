  SCRIPT campaignIntro
    ShowDlgScene
    Delay 1000
    ShowChar tutor left
    Delay 1000
    RESET tutorialSeen
    IF_TUTORIAL_SEEN
      SET tutorialSeen
    END_OF_IF
    IFNOT tutorialSeen
      Say "Ciao, novellin[g(o|a)]. Sono qui per essere il tuo istruttore di volo.                                                                       ([v(Tappa|Clicca)] dove vuoi per continuare.)"
      Ask "Allora... dimmi un po'... quanto ne sai di trasporti galattici?                                                   (Scegli una risposta.)"
    ELSE
      Say "Ciao, novellin[g(o|a)]. Sono qui per essere il tuo istruttore di volo."
      Ask "Allora... dimmi un po'... quanto ne sai di trasporti galattici?"
    END_OF_IF
    MENU
      OPT "Sono tabula rasa."
        Say "Non preoccuparti."
        Say "EssePiA S.p.A. mi ha assunto per trasformarti in un[g(|a)] camionista galattic[g(o|a)] di prim'ordine."
        Ask "Abbiamo preparato un bellissimo tutorial apposta per te."
        SET playerIsBeginner
        EXIT_MENU
      END_OF_OPT
      OPT "Conosco il gioco da tavola alla perfezione, ma questa versione digitale è nuova per me."
        Say "Oh, sarà una passeggiata. Le regole sono praticamente le stesse."
        Ask "Ed abbiamo anche un tutorial per te."
        SET playerIsAdvanced
        EXIT_MENU
      END_OF_OPT
      OPT "Dimmi solo da che parte si inizia. So già tutto!"
        Say "Oh... eccone un[g( altro|'altra)]..."
        IF_TUTORIAL_SEEN
          Say "(Un[g( altro|'altra)] che solo perchè ha già visto il tutorial una volta, pensa di sapere tutto.)"
        END_OF_IF
        SET playerIsCocky
        SET tutorialRefused
        Say "Prendi una navetta per l'Astroporto dei Novizi. Ci rivediamo lì."
        GO campaignIntroEnd
      END_OF_OPT
    END_OF_MENU
    MENU
      OPT "Figo. Sono pront[g(o|a)] per il tutorial."
        Say "Bene."
        SET tutorialAsked
        EXIT_MENU
      END_OF_OPT
      OPT_IF tutorialSeen "Beh, in realtà il Tutorial l'avrei già visto... ma rivediamolo di nuovo."
        Say "Saggia scelta."
        SET tutorialAsked
        EXIT_MENU
      END_OF_OPT
      OPT_IF tutorialSeen "Ti ringrazio, ma ho già visto questo tutorial."
        Say "Ah, ok."
        Say "(Mi sembrava in effetti un volto familiare.)"
        Say "Andiamo direttamente l'Astroporto dei Novizi, allora."
        SET tutorialSkipped
        GO campaignIntroEnd
      END_OF_OPT
      OPT_IFNOT tutorialSeen "Odio i tutorial. Dimmi solo da che parte andare."
        Say "Sei sicur[g(o|a)]?"
        Say "È il modo migliore per imparare il gioco..."
        Say "...e voleremmo insieme fino all'Astroporto dei Novizi."
        Ask "Sarà divertente."
        MENU
          OPT "Oh, va bene, mi hai convinto. Facciamolo."
            Say "Ottima scelta."
            SET tutorialAsked
            EXIT_MENU
          END_OF_OPT
          OPT "No, grazie."
            SET tutorialRefused
            Say "Come vuoi."
            Say "In questo caso, prendi una navetta per l'Astroporto dei Novizi. Ci vedremo direttamente lì."
            GO campaignIntroEnd
          END_OF_OPT
        END_OF_MENU
      END_OF_OPT
    END_OF_MENU
    LABEL campaignIntroEnd
    Delay 300
    HideChar tutor
    Delay 300
    HideDlgScene
    SUB_SCRIPT newbie_init
    IF tutorialAsked
      ACT_MISSION Tutorial
      SetCredits 0
    ELSE
      CenterMapByNode newbie
    END_OF_IF
  END_OF_SCRIPT
  SCRIPT tutorialIntro
